# build image

docker build -t kittipol/baac-api:df9a7a61 .


# run image

docker run -d -it -p 7800:8080 --name=baac-api kittipol/baac-api:df9a7a61


# remote into container

docker exec -it baac-api /bin/sh


# destroy container

docker stop baac-api; docker rm baac-api


# curl test

curl -I localhost:7800


# image export

docker save kittipol/baac-api:df9a7a61 > ./images/kittipol-baac-api-df9a7a61.tar


# load image

docker load < kittipol-baac-api-df9a7a61.tar
